{
	(nb > 1)
		rrp_generate_reward_pl	[&ITM&You obtain $nb$ $i.p$ of quality $level$ as a Ring Reward]
	(nb = 1)
		rrp_generate_reward_1	[&ITM&You obtain one $i$ of quality $level$ as a Ring Reward]	
}
